树莓派Zero2W驱动AHT20传感器并在OLED屏幕显示温湿度

2026-04-20

介绍

本文章详细记录了如何在树莓派 Zero 2 W 上通过 Python 驱动 AHT20 传感器获取温湿度，并将数据以大字体、带进度条的形式显示在 0.96 寸 OLED（SSD1306/CH1116）屏幕上。

本项目通过 Python 3 调用smbus2和Pillow库以及Gemini AI来实现。

使用说明

1. 硬件连接

AHT20 和 OLED 均使用 I2C 接口，由于树莓派支持 I2C 总线挂载多个设备，我们将它们并联连接。

模块引脚 引脚说明 树莓派引脚 (物理编号) 树莓派功能
VCC 电源 (3.3V) Pin 1 3.3V Power
GND 接地 Pin 9 Ground
SCL I2C 时钟 Pin 5 GPIO 3 (SCL)
SDA I2C 数据 Pin 3 GPIO 2 (SDA)

2. 软件配置

启用 I2C 接口

在终端输入：sudo raspi-config，依次选择 Interfacing Options -> I2C -> Yes，然后保存并退出。

安装系统字体

为了支持中文大字体显示，必须安装文泉驿正黑字体库：

import time
import os
from smbus2 import SMBus
from PIL import Image, ImageDraw, ImageFont

# --- 配置 ---
I2C_BUS = 1
AHT20_ADDR = 0x38
OLED_ADDR = 0x3C


# --- 字体加载逻辑 ---
def get_font(size):
    # 尝试多个常见路径，确保能找到中文字体
    font_paths = [
        "simhei.ttf",  # 当前目录
        "/usr/share/fonts/truetype/wqy/wqy-zenhei.ttc",  # 树莓派常见中文字体
        "/usr/share/fonts/opentype/noto/NotoSansCJK-Regular.ttc"
    ]
    for path in font_paths:
        if os.path.exists(path):
            return ImageFont.truetype(path, size)

    print(f"警告: 未找到中文字体，使用系统默认字体 (字号: {size})")
    return ImageFont.load_default()


# --- 硬件驱动 ---
def oled_command(bus, cmd):
    bus.write_byte_data(OLED_ADDR, 0x00, cmd)


def oled_init(bus):
    # 包含电荷泵开启和页寻址模式的完整序列
    cmds = [0xAE, 0xD5, 0x80, 0xA8, 0x3F, 0xD3, 0x00, 0x40, 0x8D, 0x14,
            0x20, 0x02, 0xA1, 0xC8, 0xDA, 0x12, 0x81, 0xCF, 0xD9, 0xF1,
            0xDB, 0x40, 0xA4, 0xA6, 0xAF]
    for c in cmds: oled_command(bus, c)


def oled_display(bus, image):
    img = image.convert("1")
    pix = img.load()
    for page in range(8):
        oled_command(bus, 0xB0 + page)
        oled_command(bus, 0x00)
        oled_command(bus, 0x10)
        page_data = []
        for x in range(128):
            byte = 0
            for bit in range(8):
                if pix[x, page * 8 + bit]: byte |= (1 << bit)
            page_data.append(byte)
        for i in range(0, 128, 32):
            bus.write_i2c_block_data(OLED_ADDR, 0x40, page_data[i:i + 32])


def read_aht20(bus):
    bus.write_i2c_block_data(AHT20_ADDR, 0xAC, [0x33, 0x00])
    time.sleep(0.08)
    data = bus.read_i2c_block_data(AHT20_ADDR, 0x71, 7)
    h_raw = ((data[1] << 12) | (data[2] << 4) | (data[3] >> 4))
    t_raw = ((data[3] & 0x0F) << 16) | (data[4] << 8) | data[5]
    return (t_raw / 1048576 * 200 - 50), (h_raw / 1048576 * 100)


# --- 主程序 ---
def main():
    bus = SMBus(I2C_BUS)
    oled_init(bus)

    # 加载三种字号
    f_huge = get_font(30)  # 巨大的数字
    f_mid = get_font(16)  # 中等中文
    f_small = get_font(12)  # 单位

    while True:
        try:
            temp, humi = read_aht20(bus)
            img = Image.new("1", (128, 64))
            draw = ImageDraw.Draw(img)

            # --- 1. 绘制温度区 ---
            # 数字大显示
            t_str = f"{temp:.1f}"
            draw.text((5, -2), t_str, font=f_huge, fill=255)
            # 紧跟数字后面画单位
            t_w = draw.textlength(t_str, font=f_huge)
            draw.text((5 + t_w + 2, 12), "°C", font=f_mid, fill=255)

            # --- 2. 装饰线 ---
            draw.line((0, 31, 128, 31), fill=255)

            # --- 3. 绘制湿度区 ---
            draw.text((5, 35), f"湿度:", font=f_mid, fill=255)
            draw.text((50, 35), f"{humi:.1f}%", font=f_mid, fill=255)

            # --- 4. 进度条 ---
            draw.rectangle((5, 54, 122, 60), outline=255, fill=0)
            bar_w = int(humi / 100 * 115)
            draw.rectangle((7, 56, 7 + bar_w, 58), outline=255, fill=255)

            oled_display(bus, img)
        except Exception as e:
            print(f"Error: {e}")

        time.sleep(2)


if __name__ == "__main__":
    main()
